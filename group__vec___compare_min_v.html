Поэлементный поиск минимального

Действие функции эквивалентно следующим псевдоинструкциям: 

    \~
rep N ram=[ar0++gr0];
rep N data=[ar1++gr1] with ram-data;
rep N with activate afifo;

\param nb1

    \~
\param f1cr

\~english sets activation function 
    \~
\param ar0
\~russian указатель на столбец SrcMtr1
\~english pointer to SrcMtr1 column 
    \~
\param gr0 SrcMtr1 stride

\param ar1,ar3
\~russian указатель на столбец SrcMtr2
\~english pointer to SrcMtr2 column 
    \~
\param gr1,gr3 SrcMtr2 stride

\param gr5
\~russian Высота матриц N = [0,1,2...31,32,33,...]

    \~
\param ar6
\~russian указатель на столбец DstMtr 
\~english pointer to DstMtr column 
    \~
\param gr6 DstMtr stride
